关于Vue3.0 Reactive实现原理

  • 2021-11-23

#1，在vue3.0 中 我们使用const state = reactive({ name: ‘王小二’}) 给state添加为响应式数据
#2，那么reactive 到底做了些啥呢？让我们来探讨一下

#(1).首先 我们定义一个reactive.ts 文件

#定义一个reactive方法
export function reactive(target){
//vue3.0是使用proxy 进行响应式处理
//所以首先定义个proxy代理对象
const proxy = new Proxy(target, () => {
get(target,key,recevier){
const res = Reflect.get(target, key, recevier);
return res
},
set(target,key,value,recevier){
const res = Reflect.set(target, key, value, recevier);
return res
}
})
return proxy
}

#然后引用
import { reactive } from ‘./reactive.ts’
const state = reactive({ name: ‘王小二’})

#那么reactive不是传的一共对象咋办呢 ，那set get 里面的key 不就有问题了，所以我们得加个类型判断
function isObject(value) {
return typeof value === ‘object’ && value !== null
}
export function reactive(target){
if (!isObject(target)) {
return target
}
//vue3.0是使用proxy 进行响应式处理
//所以首先定义个proxy代理对象
const proxy = new Proxy(target, () => {
get(target,key,recevier){
const res = Reflect.get(target, key, recevier);
return res
},
set(target,key,value,recevier){
const res = Reflect.set(target, key, value, recevier);
return res
}
})
return proxy
}

#那么这时又来了个问题 ，在这里调用两次同样得reactive
import { reactive } from ‘./reactive.ts’
const state1 = reactive({ name: ‘王小二’})
const state2 = reactive({ name: ‘王小二’})

#那么按上面逻辑，他两次都会去代理，这里性能不友好，所以我们需要优化下re’active方法
const reactiveMap = new WeakMap(); //是个弱引用 key 必须是对象，如果key没有被引用就会被垃圾回收
export function reactive(target){
if (!isObject(target)) {
return target
}
if(reactiveMap.get(target)){ //如果存在就直接返回了
return reactiveMap.get(target)
}
//vue3.0是使用proxy 进行响应式处理
//所以首先定义个proxy代理对象
const proxy = new Proxy(target, () => {
get(target,key,recevier){
const res = Reflect.get(target, key, recevier);
return res
},
set(target,key,value,recevier){
const res = Reflect.set(target, key, value, recevier);
return res
}
})
reactiveMap.set(target, proxy); //如果不存在就加进去
return proxy
}

这时又有个问题，我们定义了state1 和state2 ，那么 reactive(state1) == state2 吗？

答案是不相等的

所以我们需要再增加个限制

首先定义一个枚举类

const enum ReactiveFlags {
IS_REACTIVE = ‘__v_isReactive’
}
function isObject(value) {
return typeof value === ‘object’ && value !== null
}
const reactiveMap = new WeakMap();
export function reactive(target){
if(traget[ReactiveFlags.IS_REACTIVE]){ //如果这个值存在了就直接返回，再get中设置值
return target
}
if (!isObject(target)) { //是否是对象
return target
}
if(reactiveMap.get(target)){ //如果存在就直接返回了
return reactiveMap.get(target)
}
//vue3.0是使用proxy 进行响应式处理
//所以首先定义个proxy代理对象
const proxy = new Proxy(target, () => {
get(target,key,recevier){
if (key === ReactiveFlags.IS_REACTIVE) {
return true;
}
const res = Reflect.get(target, key, recevier);
return res
},
set(target,key,value,recevier){
const res = Reflect.set(target, key, value, recevier);
return res
}
})
reactiveMap.set(target, proxy); //如果不存在就加进去
return proxy
}
